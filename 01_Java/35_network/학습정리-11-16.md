### 1.캡슐화에 대하여 설명하시오.
```
	캡슐화란,
	클래스를 정의 할때,	하나의 클래스에 필요한 것들을 잘 담는 것임.
	
	(중요한 것은 적절히 담아야함.)
	
	(클래스가
		A시스템에 사용될때 역활이 달라지고
		B시스템에 사용될때 역활이 달라지므로
		적절하게 잘 담는 것은 어렵다.)
	
	클래스 캡슐화를 시켜주는 이유
		해당 클래스가 역활을 잘 담당을 해서,
		관련있는 다른 클래스들이 편하게 존재할수 있도록 하는 것
	
```

```java
package t59_server;

import java.io.DataInputStream;
import java.io.DataOutputStream;
import java.net.ServerSocket;
import java.net.Socket;
import java.util.HashMap;
import java.util.HashSet;
import java.util.Iterator;
import java.util.Scanner;
import java.util.Set;

public class Test59Server {
	public static void main(String[] args) {
		try {
			ServerSocket ss = new ServerSocket(7777);
			
			System.out.println("기달리는중");
			Socket soc = ss.accept();
			
			System.out.println("연결 받음");	
			
			Sender snd = new Sender(soc);
			Receiver rcv = new Receiver(soc);

			snd.start();
			rcv.start();
			
		} catch (Exception e) {
			e.printStackTrace();
			
		}
		
	}
}
class Sender extends Thread{
	private Socket soc;
	Sender(Socket soc){
		this.soc = soc;
	}
	public void run() {
		
		try {
			
			DataOutputStream dos = new DataOutputStream(soc.getOutputStream());
			
			while(dos!=null) {
				Scanner sc = new Scanner(System.in);
				dos.writeUTF(sc.nextLine());
			}
			
			
		} catch (Exception e) {
			e.printStackTrace();
		}
		
	}
}
class Receiver extends Thread{
	private Socket soc;
	Receiver(Socket soc){
		this.soc = soc;
	}
	public void run() {
		try {
			
			DataInputStream dis = new DataInputStream(soc.getInputStream());
			
			while(dis!=null) {
				System.out.println(dis.readUTF());
			}
			
			
		} catch (Exception e) {
			e.printStackTrace();
		}
		
	}	
}
```
